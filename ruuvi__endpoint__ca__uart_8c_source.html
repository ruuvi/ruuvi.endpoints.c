19 static uint16_t calculate_crc16 (const uint8_t * p_data,
20  const uint32_t size,
21  const uint16_t * p_crc)
22 {
23  uint32_t i;
24  uint16_t crc = RE_CA_CRC_DEFAULT;
25 
26  if ( (p_data == NULL)
27  || (size == 0))
28  {
29  crc = RE_CA_CRC_INVALID;
30  }
31  else
32  {
33  if (p_crc != NULL)
34  {
35  crc = *p_crc;
36  }
37 
38  for (i = 0; i < size; i++)
39  {
40  crc = (unsigned char) (crc >> U16_MSB_OFFSET) | (crc << U16_MSB_OFFSET);
41  crc ^= p_data[i];
42  crc ^= (unsigned char) (crc & U16_LSB_MASK) >> U8_HALF_OFFSET;
43  crc ^= (crc << U16_MSB_OFFSET) << U8_HALF_OFFSET;
44  crc ^= ( (crc & U16_LSB_MASK) << U8_HALF_OFFSET) << 1;
45  }
46  }
47 
48  return crc;
49 }

52 static void add_crc16 (uint8_t * const buffer,
53  uint32_t * written)
54 {
55  uint16_t crc16 = RE_CA_CRC_INVALID;
56  uint16_t p_crc = RE_CA_CRC_DEFAULT;
57  crc16 = calculate_crc16 (buffer + RE_CA_UART_STX_ETX_LEN,
58  (*written) - RE_CA_UART_STX_ETX_LEN, &p_crc);
59 
60  if (crc16 != RE_CA_CRC_INVALID)
61  {
62  buffer[ (*written)++] = (uint8_t) (crc16 & U16_LSB_MASK);
63  buffer[ (*written)++] = (uint8_t) ( (crc16 & U16_MSB_MASK) >> U16_MSB_OFFSET);
64  }

318 static re_status_t re_ca_uart_decode_adv_rprt (const uint8_t * const buffer,
319  re_ca_uart_payload_t * const payload)
320 {
321  re_status_t err_code = RE_SUCCESS;
322  const uint8_t adv_len = buffer[RE_CA_UART_LEN_INDEX]
323  - RE_CA_UART_MAC_BYTES
324  - RE_CA_UART_RSSI_BYTES
325  - (RE_CA_UART_MAXFIELDS) * RE_CA_UART_DELIMITER_LEN;
326  const uint8_t * const p_rssi = buffer
327  + RE_CA_UART_PAYLOAD_INDEX
328  + buffer[RE_CA_UART_LEN_INDEX]
329  - RE_CA_UART_RSSI_BYTES
330  - RE_CA_UART_DELIMITER_LEN;
331  const uint8_t * const p_data = buffer
332  + RE_CA_UART_PAYLOAD_INDEX
333  + RE_CA_UART_MAC_BYTES
334  + RE_CA_UART_DELIMITER_LEN;
335 
336  if (adv_len > RE_CA_UART_ADV_BYTES)
337  {
338  err_code |= RE_ERROR_DECODING_LEN;
339  }
340  else if (RE_CA_UART_FIELD_DELIMITER != * (p_data - RE_CA_UART_DELIMITER_LEN))
341  {
342  err_code |= RE_ERROR_DECODING_DELIMITER;
343  }
344  else if (RE_CA_UART_FIELD_DELIMITER != * (p_rssi - RE_CA_UART_DELIMITER_LEN))
345  {
346  err_code |= RE_ERROR_DECODING_DELIMITER;
347  }
348  else if (RE_CA_UART_FIELD_DELIMITER != * (p_rssi + RE_CA_UART_RSSI_BYTES))
349  {
350  err_code |= RE_ERROR_DECODING_DELIMITER;
351  }
352  else
353  {
354  memcpy (payload->params.adv.mac,
355  buffer + RE_CA_UART_PAYLOAD_INDEX,
356  RE_CA_UART_MAC_BYTES);
357  memcpy (payload->params.adv.adv,
358  p_data,
359  adv_len);
360  payload->cmd = RE_CA_UART_ADV_RPRT;
361  payload->params.adv.rssi_db = u8toi8 (*p_rssi);
362  payload->params.adv.adv_len = adv_len;
363  }
364 
365  return err_code;
366 }

368 re_status_t re_ca_uart_decode (const uint8_t * const buffer,
369  re_ca_uart_payload_t * const payload)
370 {
371  re_status_t err_code = RE_SUCCESS;
372 
373  // Sanity check buffer format
374  if (NULL == buffer)
375  {
376  err_code |= RE_ERROR_NULL;
377  }
378  else if (NULL == payload)
379  {
380  err_code |= RE_ERROR_NULL;
381  }
382  else if (RE_CA_UART_STX != buffer[RE_CA_UART_STX_INDEX])
383  {
384  err_code |= RE_ERROR_DECODING_STX;
385  }
386  else if (RE_CA_UART_ETX != buffer[buffer[RE_CA_UART_LEN_INDEX]
387  + RE_CA_UART_CRC_SIZE
388  + RE_CA_UART_HEADER_SIZE])
389  {
390  err_code |= RE_ERROR_DECODING_ETX;
391  }
392 
393 #ifdef RE_CA_SUPPORT_OLD_CMD
394  else if (RE_CA_UART_NOT_CODED != payload->cmd)
395  {
396  err_code |= RE_ERROR_INVALID_PARAM;
397  }
398 
399 #endif
400  else if (check_crc (buffer,
401  buffer[RE_CA_UART_LEN_INDEX]
402  + RE_CA_UART_HEADER_SIZE) != true)
403  {
404  err_code |= RE_ERROR_DECODING_CRC;
405  }
406  else
407  {
408  switch (buffer[RE_CA_UART_CMD_INDEX])
409  {
410 #ifdef RE_CA_SUPPORT_OLD_CMD
411 
412  case RE_CA_UART_SET_FLTR:
413  err_code |= re_ca_uart_decode_set_fltr (buffer, payload);
414  break;
415 
416  case RE_CA_UART_CLR_FLTR:
417  err_code |= re_ca_uart_decode_clr_fltr (buffer, payload);
418  break;
419 
420  case RE_CA_UART_SET_CH:
421  err_code |= re_ca_uart_decode_set_ch (buffer, payload);
422  break;
423 
424  case RE_CA_UART_SET_PHY:
425  err_code |= re_ca_uart_decode_set_phy (buffer, payload);
426  break;
427 #endif
428 
429  case RE_CA_UART_SET_FLTR_TAGS:
430  case RE_CA_UART_SET_CODED_PHY:
431  case RE_CA_UART_SET_SCAN_1MB_PHY:
432  case RE_CA_UART_SET_EXT_PAYLOAD:
433  case RE_CA_UART_SET_CH_37:
434  case RE_CA_UART_SET_CH_38:
435  case RE_CA_UART_SET_CH_39:
436  err_code |= re_ca_uart_decode_bool_payload (buffer, payload);
437  break;
438 
439  case RE_CA_UART_ACK:
440  err_code |= re_ca_uart_decode_ack (buffer, payload);
441  break;
442 
443  case RE_CA_UART_SET_FLTR_ID:
444  err_code |= re_ca_uart_decode_fltr_id (buffer, payload);
445  break;
446 
447  case RE_CA_UART_SET_ALL:
448  err_code |= re_ca_uart_decode_all (buffer, payload);

450 
451  case RE_CA_UART_ADV_RPRT:
452  err_code |= re_ca_uart_decode_adv_rprt (buffer, payload);

454

458  }
459  }

465  uint8_t * const buf_len,

467 {

469  uint32_t written = 0;
470 
471  if (RE_CA_UART_TX_MAX_LEN > *buf_len)
472  {
473  err_code |= RE_ERROR_DATA_SIZE;
474  }

480  + payload->params.adv.adv_len

482  + RE_CA_UART_ADV_FIELDS * RE_CA_UART_DELIMITER_LEN;
483  buffer[RE_CA_UART_CMD_INDEX] = RE_CA_UART_ADV_RPRT;
484  written += RE_CA_UART_HEADER_SIZE;
485  memcpy (buffer + RE_CA_UART_PAYLOAD_INDEX,
486  payload->params.adv.mac,
487  RE_CA_UART_MAC_BYTES);
488  written += RE_CA_UART_MAC_BYTES;
489  buffer[written++] = RE_CA_UART_FIELD_DELIMITER;
490  memcpy (buffer + written,
491  payload->params.adv.adv,
492  payload->params.adv.adv_len);
493  written += payload->params.adv.adv_len;
494  buffer[written++] = RE_CA_UART_FIELD_DELIMITER;
495  buffer[written++] = i8tou8 (payload->params.adv.rssi_db);
496  buffer[written++] = RE_CA_UART_FIELD_DELIMITER;
497  add_crc16 (buffer, &written);
498  buffer[written++] = RE_CA_UART_ETX;
499  *buf_len = written;
500  }
501 
502  return err_code;

598 static re_status_t re_ca_uart_encode_all (uint8_t * const buffer,
599  uint8_t * const buf_len,
600  const re_ca_uart_payload_t * const payload)
601 {

603  uint32_t written = 0;
604 
605  if (RE_CA_UART_TX_MAX_LEN > *buf_len)
606  {
607  err_code |= RE_ERROR_DATA_SIZE;
608  }
609  else
610  {
611  buffer[RE_CA_UART_STX_INDEX] = RE_CA_UART_STX;
612  // Payload length is different from total message length.
613  buffer[RE_CA_UART_LEN_INDEX] = RE_CA_UART_CMD_ALL_BOOL_LEN
614  + RE_CA_UART_CMD_FLTR_ID_LEN
615  + (RE_CA_UART_ALL_FIELDS * RE_CA_UART_DELIMITER_LEN);
616  buffer[RE_CA_UART_CMD_INDEX] = payload->cmd;
617  written += RE_CA_UART_HEADER_SIZE;
618  memcpy (buffer + RE_CA_UART_PAYLOAD_INDEX,
619  (void *) &payload->params.fltr_id_param.id,
620  RE_CA_UART_CMD_FLTR_ID_LEN);
621  written += RE_CA_UART_CMD_FLTR_ID_LEN;
622  buffer[written++] = RE_CA_UART_FIELD_DELIMITER;
623  buffer[written] = ( ( (payload->params.all_params.bools.fltr_tags.state & 1U)

625  | ( (payload->params.all_params.bools.coded_phy.state & 1U)

647 re_status_t re_ca_uart_encode (uint8_t * const buffer, uint8_t * const buf_len,
648  const re_ca_uart_payload_t * const payload)
649 {
650  re_status_t err_code = RE_SUCCESS;
651 
652  if (NULL == buffer)
653  {
654  err_code |= RE_ERROR_NULL;
655  }
656  else if (NULL == buf_len)
657  {
658  err_code |= RE_ERROR_NULL;
659  }
660  else if (NULL == payload)
661  {
662  err_code |= RE_ERROR_NULL;
663  }
664  else
665  {
666  switch (payload->cmd)
667  {
668  case RE_CA_UART_SET_FLTR_TAGS:
669  case RE_CA_UART_SET_CODED_PHY:
670  case RE_CA_UART_SET_SCAN_1MB_PHY:
671  case RE_CA_UART_SET_EXT_PAYLOAD:
672  case RE_CA_UART_SET_CH_37:
673  case RE_CA_UART_SET_CH_38:
674  case RE_CA_UART_SET_CH_39:
675  err_code |= re_ca_uart_encode_bool_payload (buffer, buf_len, payload);
676  break;
677 
678  case RE_CA_UART_ACK:
679  err_code |= re_ca_uart_encode_ack (buffer, buf_len, payload);
680  break;
681 
682  case RE_CA_UART_SET_FLTR_ID:
683  err_code |= re_ca_uart_encode_fltr_id (buffer, buf_len, payload);
684  break;
685 
686  case RE_CA_UART_SET_ALL:
687  err_code |= re_ca_uart_encode_all (buffer, buf_len, payload);
688  break;
689 
690  case RE_CA_UART_ADV_RPRT:
691  err_code |= re_ca_uart_encode_adv_rprt (buffer, buf_len, payload);
692  break;
693 
694  default:
695  err_code |= RE_ERROR_INVALID_PARAM;
696  break;
697  }
