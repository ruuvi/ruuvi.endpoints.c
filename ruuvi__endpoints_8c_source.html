44uint8_t re_calc_crc8 (const uint8_t * DataArray, const uint16_t Length)
45{
46 uint16_t i;
47 uint8_t CRC;
48 CRC = 0;
49
50 for (i = 0; i < Length; i++)
51 { CRC = CRC_8_TABLE[CRC ^ DataArray[i]]; }
52
53 return CRC;
54}

150static int32_t f2i (re_float value)
151{
152 int32_t rvalue = 0x80000000;
153 value = (value) >= 0 ? (value) + 0.5f : (value) - 0.5f;
154
155 if (value >= INT32_MAX)
156 {
157 rvalue = INT32_MAX;
158 }
159 else if (value <= (INT32_MIN + 1))
160 {
161 rvalue = INT32_MIN + 1;
162 }
163 else
164 {
165 rvalue = (int32_t) value;
166 }
167
168 return rvalue;
169}

185re_status_t re_log_write_data (uint8_t * const buffer, const re_float data,
186 const uint8_t source)
187{
188 re_status_t err_code = RE_SUCCESS;
189 int32_t discrete_value = 0xFFFFFFFF;
190 re_float scaled_value = 0;
191
192 if (NULL == buffer)
193 {
194 err_code |= RE_ERROR_NULL;
195 }
196 else if (isnan (data) || isinf (data))
197 {
198 err_code |= RE_ERROR_INVALID_PARAM;
199 }
200 else
201 {
202 switch (source)
203 {
204 case RE_STANDARD_DESTINATION_ACCELERATION_X:
205 case RE_STANDARD_DESTINATION_ACCELERATION_Y:
206 case RE_STANDARD_DESTINATION_ACCELERATION_Z:
207 scaled_value = (data * RE_STANDARD_ACCELERATION_SF);
208 break;
209
210 case RE_STANDARD_DESTINATION_GYRATION_X:
211 case RE_STANDARD_DESTINATION_GYRATION_Y:
212 case RE_STANDARD_DESTINATION_GYRATION_Z:
213 scaled_value = (data * RE_STANDARD_GYRATION_SF);
214 break;
215
216 case RE_STANDARD_DESTINATION_HUMIDITY:
217 scaled_value = (data * RE_STANDARD_HUMIDITY_SF);
218 break;
219
220 case RE_STANDARD_DESTINATION_PRESSURE:
221 scaled_value = (data * RE_STANDARD_PRESSURE_SF);
222 break;
223
224 case RE_STANDARD_DESTINATION_TEMPERATURE:
225 scaled_value = (data * RE_STANDARD_TEMPERATURE_SF);
226 break;
227
228 default:
229 err_code |= RE_ERROR_NOT_IMPLEMENTED;
230 break;
231 }
232
233 scaled_value = roundf (scaled_value);
234 discrete_value = f2i (scaled_value);
235 }
236
237 // These shifts do not rely on the value of leftmost bit if original
238 // value is negative, so this is safe way to encode bytes.
239 buffer[RE_LOG_WRITE_VALUE_MSB_IDX] = (uint8_t) ( (discrete_value >> 24U) & 0xFFU);
240 buffer[RE_LOG_WRITE_VALUE_B2_IDX] = (uint8_t) ( (discrete_value >> 16U) & 0xFFU);
241 buffer[RE_LOG_WRITE_VALUE_B3_IDX] = (uint8_t) ( (discrete_value >> 8U) & 0xFFU);
242 buffer[RE_LOG_WRITE_VALUE_LSB_IDX] = (uint8_t) (discrete_value & 0xFFU);
243 return err_code;
244}
