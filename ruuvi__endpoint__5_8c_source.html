50  float acceleration)

53

56  re_clip (&acceleration, RE_5_ACC_MIN, RE_5_ACC_MAX);
57  int16_t rounded_acceleration = (int16_t) roundf (acceleration * RE_5_ACC_RATIO);
58  // Type cast adds 2^16 to a negative signed value, not changing bits.
59  coded_acceleration = (uint16_t) rounded_acceleration;
