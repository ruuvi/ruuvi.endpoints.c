165  const re_8_data_t * const data,
166  re_8_encrypt_fp cipher,

168  const size_t key_size)
169 {

171 
172  if ( (NULL == buffer) || (NULL == data) || (NULL == cipher) || (NULL == key))
173  {
174  result |= RE_ERROR_NULL;
175  }
176  else
177  {
178  uint32_t cipher_status = 0;
179  uint8_t cleartext[RE_8_CIPHERTEXT_LENGTH] = {0};

181  re_8_encode_humidity (buffer, data);

187  re_8_encode_reserved (buffer);
188  re_8_encode_crc (buffer);

190  cipher_status = cipher (cleartext, buffer + RE_8_OFFSET_CIPHER,
191  RE_8_CIPHERTEXT_LENGTH, key, key_size);
192  re_8_encode_set_address (buffer, data);
193 
194  if (0 != cipher_status)
