183static void
184re_7_encode_tilt (uint8_t * const buffer, const re_7_data_t * data)
185{
186 int8_t coded_tilt_x = RE_7_INVALID_TILT;
187 int8_t coded_tilt_y = RE_7_INVALID_TILT;
188 const re_float acc_x = data->acceleration_x_g;
189 const re_float acc_y = data->acceleration_y_g;
190 const re_float acc_z = data->acceleration_z_g;
191
192 /* Check if any acceleration component is valid */
193 if ( (!isnan (acc_x)) && (!isnan (acc_y)) && (!isnan (acc_z)))
194 {
195 /* Calculate magnitude of acceleration vector */
196 const re_float magnitude = sqrtf (acc_x * acc_x + acc_y * acc_y + acc_z * acc_z);
197
198 /* Avoid division by zero */
199 if (magnitude > 0.001f)
200 {
201 /* Normalize and calculate tilt angles using asin */
202 /* tilt = asin(acc_component / magnitude) converted to degrees */
203 re_float norm_x = acc_x / magnitude;
204 re_float norm_y = acc_y / magnitude;
205
206 /* Clamp to valid asin range [-1, 1] to handle floating point errors */
207 if (norm_x > 1.0f)
208 {
209 norm_x = 1.0f;
210 }
211
212 if (norm_x < -1.0f)
213 {
214 norm_x = -1.0f;
215 }
216
217 if (norm_y > 1.0f)
218 {
219 norm_y = 1.0f;
220 }
221
222 if (norm_y < -1.0f)
223 {
224 norm_y = -1.0f;
225 }
226
227 re_float tilt_x = asinf (norm_x) * RE_7_RAD_TO_DEG;
228 re_float tilt_y = asinf (norm_y) * RE_7_RAD_TO_DEG;
229 re_clip (&tilt_x, RE_7_TILT_MIN, RE_7_TILT_MAX);
230 re_clip (&tilt_y, RE_7_TILT_MIN, RE_7_TILT_MAX);
231 coded_tilt_x = (int8_t) lrintf (tilt_x * RE_7_TILT_RATIO);
232 coded_tilt_y = (int8_t) lrintf (tilt_y * RE_7_TILT_RATIO);
233 }
234 }
235
236 buffer[RE_7_OFFSET_TILT_X] = (uint8_t) coded_tilt_x;
237 buffer[RE_7_OFFSET_TILT_Y] = (uint8_t) coded_tilt_y;
238}

491re_status_t
492re_7_decode (const uint8_t * const p_buffer, re_7_data_t * const p_data)
493{
494 const uint8_t * const p_payload = &p_buffer[RE_7_OFFSET_PAYLOAD];
495 re_status_t result = RE_SUCCESS;
496
497 if ( (NULL == p_buffer) || (NULL == p_data))
498 {
499 return RE_ERROR_NULL;
500 }
501
502 memset (p_data, 0, sizeof (*p_data));
503
504 if (RE_7_DESTINATION != p_payload[RE_7_OFFSET_HEADER])
505 {
506 return RE_ERROR_INVALID_PARAM;
507 }
508
509 /* Verify CRC */
510 const uint8_t calculated_crc = re_calc_crc8 (p_payload, RE_7_CRC_DATA_LEN);
511
512 if (calculated_crc != p_payload[RE_7_OFFSET_CRC])
513 {
514 return RE_ERROR_DECODING_CRC;
515 }
516
517 p_data->sequence_counter = re_7_decode_sequence (p_payload);
518 re_7_decode_flags (p_payload, p_data);
519 p_data->temperature_c = re_7_decode_temperature (p_payload);
520 p_data->humidity_rh = re_7_decode_humidity (p_payload);
521 p_data->pressure_pa = re_7_decode_pressure (p_payload);
522 re_7_decode_tilt (p_payload, &p_data->tilt_x_deg, &p_data->tilt_y_deg);
523 p_data->luminosity_lux = re_7_decode_luminosity (p_payload);
524 p_data->color_temp_k = re_7_decode_color_temp (p_payload);
525 re_7_decode_batt_motion (p_payload, &p_data->battery_v, &p_data->motion_intensity);
526 p_data->motion_count = re_7_decode_motion_count (p_payload);
527 p_data->address = re_7_decode_mac (p_payload);
528 return result;
529}
