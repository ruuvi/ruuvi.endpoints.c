207static inline void
208re_encode_u24 (
209 uint8_t * const p_slot,
210 const re_float val,
211 const uint32_t invalid_val,
212 const re_float min_val,
213 const re_float max_val,
214 const re_float scale_factor)
215{
216 uint32_t coded_val = invalid_val;
217
218 if (!isnan (val))
219 {
220 const re_float val_clipped = RE_CLIP (val, min_val, max_val);
221 coded_val = (uint32_t) lrintf ( (val_clipped - min_val) * scale_factor);
222 }
223
224 p_slot[0] |= re_be24_get_high_byte (coded_val);
225 p_slot[1] |= re_be24_get_mid_byte (coded_val);
226 p_slot[2] |= re_be24_get_low_byte (coded_val);
227}

229static inline re_float
230re_decode_u24 (
231 const uint8_t * const p_slot,
232 const uint32_t invalid_val,
233 const re_float min_val,
234 const re_float scale_factor)
235{
236 uint32_t coded_val = 0;
237 coded_val |= ( (uint32_t) p_slot[0]) << RE_BYTE_2_SHIFT;
238 coded_val |= ( (uint32_t) p_slot[1]) << RE_BYTE_1_SHIFT;
239 coded_val |= ( (uint32_t) p_slot[2]) << RE_BYTE_0_SHIFT;
240
241 if (invalid_val == coded_val)
242 {
243 return NAN;
244 }
245
246 return (re_float) coded_val / scale_factor + min_val;
247}

348static void
349re_e1_encode_address (uint8_t * const p_buffer, const uint64_t mac_addr)
350{
351 // Address is 64 bits, skip 2 first bytes
352 uint8_t addr_offset = RE_E1_OFFSET_ADDR_MSB;
353 uint64_t mac = mac_addr;
354#pragma GCC diagnostic push
355#pragma GCC diagnostic ignored "-Wtype-limits"
356
357 if (RE_E1_MAC_MAX < mac_addr)
358 {
359 mac = RE_E1_INVALID_MAC;
360 }
361 else if (RE_E1_MAC_MIN > mac_addr)
362 {
363 mac = RE_E1_INVALID_MAC;
364 }
365 else
366 {
367 // MISRA C:2012, 15.7 - All if...else if constructs shall be terminated with an else statement
368 }
369
370#pragma GCC diagnostic pop
371 p_buffer[addr_offset] = (mac >> RE_BYTE_5_SHIFT) & RE_BYTE_MASK;
372 addr_offset++;
373 p_buffer[addr_offset] = (mac >> RE_BYTE_4_SHIFT) & RE_BYTE_MASK;
374 addr_offset++;
375 p_buffer[addr_offset] = (mac >> RE_BYTE_3_SHIFT) & RE_BYTE_MASK;
376 addr_offset++;
377 p_buffer[addr_offset] = (mac >> RE_BYTE_2_SHIFT) & RE_BYTE_MASK;
378 addr_offset++;
379 p_buffer[addr_offset] = (mac >> RE_BYTE_1_SHIFT) & RE_BYTE_MASK;
380 addr_offset++;
381 p_buffer[addr_offset] = (mac >> RE_BYTE_0_SHIFT) & RE_BYTE_MASK;
382}

385re_e1_decode_address (const uint8_t * const p_buffer)
386{
387 // Address is 64 bits, skip 2 first bytes
388 uint8_t addr_offset = 0;
389 uint64_t mac = 0;
390 mac |= p_buffer[addr_offset];
391 mac <<= RE_BYTE_1_SHIFT;
392 addr_offset++;
393 mac |= p_buffer[addr_offset];
394 mac <<= RE_BYTE_1_SHIFT;
395 addr_offset++;
396 mac |= p_buffer[addr_offset];
397 mac <<= RE_BYTE_1_SHIFT;
398 addr_offset++;
399 mac |= p_buffer[addr_offset];
400 mac <<= RE_BYTE_1_SHIFT;
401 addr_offset++;
402 mac |= p_buffer[addr_offset];
403 mac <<= RE_BYTE_1_SHIFT;
404 addr_offset++;
405 mac |= p_buffer[addr_offset];
406 return mac;

409re_status_t
410re_e1_encode (uint8_t * const p_buffer, const re_e1_data_t * const p_data)
411{
412 re_status_t result = RE_SUCCESS;
413
414 if ( (NULL == p_buffer) || (NULL == p_data))
415 {
416 result |= RE_ERROR_NULL;
417 }
418 else
419 {
420 memset (p_buffer, 0x00U, RE_E1_DATA_LENGTH);
421 p_buffer[RE_E1_OFFSET_HEADER] = RE_E1_DESTINATION;
422 re_e1_encode_temperature (&p_buffer[RE_E1_OFFSET_TEMPERATURE_MSB], p_data->temperature_c);
423 re_e1_encode_humidity (&p_buffer[RE_E1_OFFSET_HUMIDITY_MSB], p_data->humidity_rh);
424 re_e1_encode_pressure (&p_buffer[RE_E1_OFFSET_PRESSURE_MSB], p_data->pressure_pa);
425 re_e1_encode_pm (&p_buffer[RE_E1_OFFSET_PM_1_0_MSB], p_data->pm1p0_ppm);
426 re_e1_encode_pm (&p_buffer[RE_E1_OFFSET_PM_2_5_MSB], p_data->pm2p5_ppm);
427 re_e1_encode_pm (&p_buffer[RE_E1_OFFSET_PM_4_0_MSB], p_data->pm4p0_ppm);
428 re_e1_encode_pm (&p_buffer[RE_E1_OFFSET_PM_10_0_MSB], p_data->pm10p0_ppm);
429 re_e1_encode_co2 (&p_buffer[RE_E1_OFFSET_CO2_MSB], p_data->co2);
430 re_e1_encode_voc (&p_buffer[RE_E1_OFFSET_VOC], &p_buffer[RE_E1_OFFSET_FLAGS],
431 p_data->voc);
432 re_e1_encode_nox (&p_buffer[RE_E1_OFFSET_NOX], &p_buffer[RE_E1_OFFSET_FLAGS],
433 p_data->nox);
434 re_e1_encode_luminosity (&p_buffer[RE_E1_OFFSET_LUMINOSITY_MSB], p_data->luminosity);
435 re_e1_encode_sound_dba (
436 &p_buffer[RE_E1_OFFSET_SOUND_DBA_INST],
437 &p_buffer[RE_E1_OFFSET_FLAGS],
438 &re_e1_coeffs_sound_dba_inst,
439 p_data->sound_dba_inst);
440 re_e1_encode_sound_dba (
441 &p_buffer[RE_E1_OFFSET_SOUND_DBA_AVG],
442 &p_buffer[RE_E1_OFFSET_FLAGS],
443 &re_e1_coeffs_sound_dba_avg,
444 p_data->sound_dba_avg);
445 re_e1_encode_sound_dba (
446 &p_buffer[RE_E1_OFFSET_SOUND_DBA_PEAK],
447 &p_buffer[RE_E1_OFFSET_FLAGS],
448 &re_e1_coeffs_sound_dba_peak,
449 p_data->sound_dba_peak);
450 re_e1_encode_sequence (&p_buffer[RE_E1_OFFSET_SEQ_CNT_MSB], p_data->seq_cnt);
451 re_e1_encode_flags (&p_buffer[RE_E1_OFFSET_FLAGS], p_data->flags);
452 p_buffer[RE_E1_OFFSET_RESERVED + 0] = 0xFFU;
453 p_buffer[RE_E1_OFFSET_RESERVED + 1] = 0xFFU;
454 p_buffer[RE_E1_OFFSET_RESERVED + 2] = 0xFFU;
455 p_buffer[RE_E1_OFFSET_RESERVED + 3] = 0xFFU;
456 p_buffer[RE_E1_OFFSET_RESERVED + 4] = 0xFFU;
457 p_buffer[RE_E1_OFFSET_RESERVED + 5] = 0xFFU;
458 re_e1_encode_address (p_buffer, p_data->address);
459 }
460
461 return result;
462}

464bool
465re_e1_check_format (const uint8_t * const p_buffer)
466{
467 if (NULL == p_buffer)
468 {
469 return false;
470 }
471
472 if (RE_E1_RAW_PACKET_LENGTH_VAL != p_buffer[RE_E1_RAW_PACKET_LENGTH_OFFSET])
473 {
474 return false;
475 }
476
477 if (RE_E1_RAW_PACKET_TYPE_VAL != p_buffer[RE_E1_RAW_PACKET_TYPE_OFFSET])
478 {
479 return false;
480 }
481
482 const uint16_t manufacturer_id = (uint16_t) ( (uint16_t)
483 p_buffer[RE_E1_RAW_PACKET_MANUFACTURER_ID_OFFSET_HI]
484 << RE_BYTE_1_SHIFT)
485 + p_buffer[RE_E1_RAW_PACKET_MANUFACTURER_ID_OFFSET_LO];
486
487 if (RE_E1_RAW_PACKET_MANUFACTURER_ID_VAL != manufacturer_id)
488 {
489 return false;
490 }
491
492 if (RE_E1_DESTINATION != p_buffer[RE_E1_OFFSET_PAYLOAD + RE_E1_OFFSET_HEADER])
493 {
494 return false;
495 }
496
497 return true;
498}

500re_status_t
501re_e1_decode (const uint8_t * const p_buffer, re_e1_data_t * const p_data)
502{
503 const uint8_t * const p_payload = &p_buffer[RE_E1_OFFSET_PAYLOAD];
504 re_status_t result = RE_SUCCESS;
505
506 if ( (NULL == p_payload) || (NULL == p_data))
507 {
508 return RE_ERROR_NULL;
509 }
510
511 memset (p_data, 0, sizeof (*p_data));
512
513 if (RE_E1_DESTINATION != p_payload[RE_E1_OFFSET_HEADER])
514 {
515 return RE_ERROR_INVALID_PARAM;
516 }
517
518 p_data->temperature_c = re_e1_decode_temperature (
519 &p_payload[RE_E1_OFFSET_TEMPERATURE_MSB]);
520 p_data->humidity_rh = re_e1_decode_humidity (&p_payload[RE_E1_OFFSET_HUMIDITY_MSB]);
521 p_data->pressure_pa = re_e1_decode_pressure (&p_payload[RE_E1_OFFSET_PRESSURE_MSB]);
522 p_data->pm1p0_ppm = re_e1_decode_pm (&p_payload[RE_E1_OFFSET_PM_1_0_MSB]);
523 p_data->pm2p5_ppm = re_e1_decode_pm (&p_payload[RE_E1_OFFSET_PM_2_5_MSB]);
524 p_data->pm4p0_ppm = re_e1_decode_pm (&p_payload[RE_E1_OFFSET_PM_4_0_MSB]);
525 p_data->pm10p0_ppm = re_e1_decode_pm (&p_payload[RE_E1_OFFSET_PM_10_0_MSB]);
526 p_data->co2 = re_e1_decode_co2 (&p_payload[RE_E1_OFFSET_CO2_MSB]);
527 p_data->voc = re_e1_decode_voc (&p_payload[RE_E1_OFFSET_VOC],
528 &p_payload[RE_E1_OFFSET_FLAGS]);
529 p_data->nox = re_e1_decode_nox (&p_payload[RE_E1_OFFSET_NOX],
530 &p_payload[RE_E1_OFFSET_FLAGS]);
531 p_data->luminosity = re_e1_decode_luminosity (
532 &p_payload[RE_E1_OFFSET_LUMINOSITY_MSB]);
533 p_data->sound_dba_inst = re_e1_decode_sound_dba (
534 &p_payload[RE_E1_OFFSET_SOUND_DBA_INST],
535 &p_payload[RE_E1_OFFSET_FLAGS],
536 &re_e1_coeffs_sound_dba_inst);
537 p_data->sound_dba_avg = re_e1_decode_sound_dba (
538 &p_payload[RE_E1_OFFSET_SOUND_DBA_AVG],
539 &p_payload[RE_E1_OFFSET_FLAGS],
540 &re_e1_coeffs_sound_dba_avg);
541 p_data->sound_dba_peak = re_e1_decode_sound_dba (
542 &p_payload[RE_E1_OFFSET_SOUND_DBA_PEAK],
543 &p_payload[RE_E1_OFFSET_FLAGS],
544 &re_e1_coeffs_sound_dba_peak);
545 p_data->seq_cnt = re_e1_decode_sequence (&p_payload[RE_E1_OFFSET_SEQ_CNT_MSB]);
546 p_data->flags = re_e1_decode_flags (&p_payload[RE_E1_OFFSET_FLAGS]);
547 p_data->address = re_e1_decode_address (&p_payload[RE_E1_OFFSET_ADDR_MSB]);
548 return result;
549}
