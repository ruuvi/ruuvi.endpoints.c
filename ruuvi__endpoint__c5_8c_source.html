53static void re_c5_encode_set_address (uint8_t * const buffer, const re_c5_data_t * data)
54{
55 // Address is 64 bits, skip 2 first bytes
56 uint8_t addr_offset = RE_C5_OFFSET_ADDR_MSB;
57 uint64_t mac = data->address;
58
59 if (RE_C5_MAC_MAX < data->address)
60 {
61 mac = RE_C5_INVALID_MAC;
62 }
63
64#pragma GCC diagnostic push
65#pragma GCC diagnostic ignored "-Wtype-limits"
66
67 if (RE_C5_MAC_MIN > data->address) // cppcheck-suppress unsignedLessThanZero
68 {
69 mac = RE_C5_INVALID_MAC;
70 }
71
72#pragma GCC diagnostic pop
73 buffer[addr_offset] = (mac >> RE_C5_BYTE_5_SHIFT) & RE_C5_BYTE_MASK;
74 addr_offset++;
75 buffer[addr_offset] = (mac >> RE_C5_BYTE_4_SHIFT) & RE_C5_BYTE_MASK;
76 addr_offset++;
77 buffer[addr_offset] = (mac >> RE_C5_BYTE_3_SHIFT) & RE_C5_BYTE_MASK;
78 addr_offset++;
79 buffer[addr_offset] = (mac >> RE_C5_BYTE_2_SHIFT) & RE_C5_BYTE_MASK;
80 addr_offset++;
81 buffer[addr_offset] = (mac >> RE_C5_BYTE_1_SHIFT) & RE_C5_BYTE_MASK;
82 addr_offset++;
83 buffer[addr_offset] = (mac >> 0) & RE_C5_BYTE_MASK;
84}

86static void re_c5_encode_humidity (uint8_t * const buffer, const re_c5_data_t * data)
87{
88 uint16_t coded_humidity = RE_C5_INVALID_HUMIDITY;
89 re_float humidity = data->humidity_rh;
90
91 if (!isnan (humidity))
92 {
93 re_clip (&humidity, RE_C5_HUMI_MIN, RE_C5_HUMI_MAX);
94 coded_humidity = (uint16_t) roundf (humidity * RE_C5_HUMI_RATIO);
95 }
96
97 buffer[RE_C5_OFFSET_HUMI_MSB] = coded_humidity >> RE_C5_BYTE_1_SHIFT;
98 buffer[RE_C5_OFFSET_HUMI_LSB] = coded_humidity & RE_C5_BYTE_MASK;
99}

101static void re_c5_encode_temperature (uint8_t * const buffer, const re_c5_data_t * data)
102{
103 uint16_t coded_temperature = RE_C5_INVALID_TEMPERATURE;
104 re_float temperature = data->temperature_c;
105
106 if (!isnan (temperature))
107 {
108 re_clip (&temperature, RE_C5_TEMP_MIN, RE_C5_TEMP_MAX);
109 int16_t rounded_temperature = (int16_t) roundf (temperature * RE_C5_TEMP_RATIO);
110 // Type cast adds 2^16 to a negative signed value, not changing bits.
111 coded_temperature = (uint16_t) rounded_temperature;
112 }
113
114 buffer[RE_C5_OFFSET_TEMP_MSB] = coded_temperature >> RE_C5_BYTE_1_SHIFT;
115 buffer[RE_C5_OFFSET_TEMP_LSB] = coded_temperature & RE_C5_BYTE_MASK;
116}

135static void re_c5_encode_pwr (uint8_t * const buffer, const re_c5_data_t * data)
136{
137 uint16_t coded_voltage = RE_C5_INVALID_VOLTAGE;
138 re_float voltage = data->battery_v;
139 uint16_t coded_tx_power = RE_C5_INVALID_POWER;
140 re_float tx_power = (re_float) data->tx_power;
141
142 if (!isnan (voltage))
143 {
144 re_clip (&voltage, RE_C5_VOLTAGE_MIN, RE_C5_VOLTAGE_MAX);
145 coded_voltage = (uint16_t) roundf ( (voltage * RE_C5_BATT_RATIO)
146 - RE_C5_BATT_OFFSET);
147 }
148
149 // Check against original int value
150 if (RE_C5_INVALID_POWER != data->tx_power)
151 {
152 re_clip (&tx_power, RE_C5_TXPWR_MIN, RE_C5_TXPWR_MAX);
153 coded_tx_power = (uint16_t) roundf ( (tx_power
154 + RE_C5_TXPWR_OFFSET)
155 / RE_C5_TXPWR_RATIO);
156 }
157
158 uint16_t power_info = ( (uint16_t) (coded_voltage << RE_C5_BYTE_VOLTAGE_OFFSET))
159 + coded_tx_power;
160 buffer[RE_C5_OFFSET_POWER_MSB] = (power_info >> RE_C5_BYTE_1_SHIFT);
161 buffer[RE_C5_OFFSET_POWER_LSB] = (power_info & RE_C5_BYTE_MASK);
162}

189re_status_t re_c5_encode (uint8_t * const buffer, const re_c5_data_t * data)
190{
191 re_status_t result = RE_SUCCESS;
192
193 if ( (NULL == buffer) || (NULL == data))
194 {
195 result |= RE_ERROR_NULL;
196 }
197 else
198 {
199 buffer[RE_C5_OFFSET_HEADER] = RE_C5_DESTINATION;
200 re_c5_encode_humidity (buffer, data);
201 re_c5_encode_temperature (buffer, data);
202 re_c5_encode_pressure (buffer, data);
203 re_c5_encode_movement (buffer, data);
204 re_c5_encode_sequence (buffer, data);
205 re_c5_encode_pwr (buffer, data);
206 re_c5_encode_set_address (buffer, data);
207 }
208
209 return result;
210}
